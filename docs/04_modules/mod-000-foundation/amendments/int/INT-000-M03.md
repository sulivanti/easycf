### Novos paths em INT-004 — Users API

Adicionar ao catálogo OpenAPI da Users API (INT-004):

#### POST /api/v1/users/{id}/reset-password

```yaml
/api/v1/users/{id}/reset-password:
  post:
    operationId: users_reset_password
    summary: Admin inicia reset de senha para o usuário
    tags: [Users]
    security:
      - bearerAuth: []
    parameters:
      - name: id
        in: path
        required: true
        schema: { type: string, format: uuid }
    requestBody:
      required: false
      content:
        application/json:
          schema:
            type: object
            description: "Body vazio ({})"
    responses:
      '200':
        description: E-mail de reset enviado com sucesso
        content:
          application/json:
            schema:
              $ref: '#/components/schemas/ResetPasswordResponse'
      '401':
        description: Sessão inválida ou ausente
        content:
          application/json:
            schema:
              $ref: '#/components/schemas/ProblemDetails'
      '403':
        description: Scope users:user:write ausente
        content:
          application/json:
            schema:
              $ref: '#/components/schemas/ProblemDetails'
      '404':
        description: Usuário não encontrado
        content:
          application/json:
            schema:
              $ref: '#/components/schemas/ProblemDetails'
      '500':
        description: Erro interno
        content:
          application/json:
            schema:
              $ref: '#/components/schemas/ProblemDetails'
```

#### DELETE /api/v1/users/{id}/invite

```yaml
/api/v1/users/{id}/invite:
  delete:
    operationId: users_cancel_invite
    summary: Cancela convite pendente de ativação
    tags: [Users]
    security:
      - bearerAuth: []
    parameters:
      - name: id
        in: path
        required: true
        schema: { type: string, format: uuid }
    responses:
      '200':
        description: Convite cancelado, status alterado para INACTIVE
        content:
          application/json:
            schema:
              $ref: '#/components/schemas/CancelInviteResponse'
      '401':
        description: Sessão inválida ou ausente
        content:
          application/json:
            schema:
              $ref: '#/components/schemas/ProblemDetails'
      '403':
        description: Scope users:user:delete ausente
        content:
          application/json:
            schema:
              $ref: '#/components/schemas/ProblemDetails'
      '404':
        description: Usuário não encontrado
        content:
          application/json:
            schema:
              $ref: '#/components/schemas/ProblemDetails'
      '409':
        description: Usuário não está em status PENDING
        content:
          application/json:
            schema:
              $ref: '#/components/schemas/ProblemDetails'
      '500':
        description: Erro interno
        content:
          application/json:
            schema:
              $ref: '#/components/schemas/ProblemDetails'
```

### Novos schemas

```yaml
ResetPasswordResponse:
  type: object
  required: [message]
  properties:
    message:
      type: string
      example: "Password reset email sent."

CancelInviteResponse:
  type: object
  required: [data]
  properties:
    data:
      type: object
      required: [id, status, updatedAt]
      properties:
        id: { type: string, format: uuid }
        status: { type: string, enum: [INACTIVE] }
        updatedAt: { type: string, format: date-time }
```
